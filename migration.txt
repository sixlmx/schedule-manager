BEGIN;

CREATE EXTENSION IF NOT EXISTS pgcrypto;

CREATE TABLE IF NOT EXISTS public.teachers (
  id integer GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  name text,
  fio text,
  "position" text,
  color text
);

CREATE TABLE IF NOT EXISTS public.subjects (
  id integer GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  name text NOT NULL,
  abbreviation text
);

CREATE TABLE IF NOT EXISTS public.groups (
  id integer GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  name text NOT NULL,
  abbreviation text,
  year_of_admission text
);

CREATE TABLE IF NOT EXISTS public.bells (
  id integer GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  lesson_number integer NOT NULL UNIQUE,
  start_time time without time zone NOT NULL,
  end_time time without time zone NOT NULL
);

CREATE TABLE IF NOT EXISTS public.lessons (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  weekday integer NOT NULL CHECK (weekday >= 1 AND weekday <= 7),
  lesson_number integer NOT NULL,
  start_time text NOT NULL,
  end_time text NOT NULL,
  start_time_min integer NOT NULL,
  end_time_min integer NOT NULL,
  type_lesson text,
  cabinet text,
  subject_id integer REFERENCES public.subjects(id) ON DELETE SET NULL
);

CREATE TABLE IF NOT EXISTS public.lesson_teachers (
  lesson_id uuid NOT NULL REFERENCES public.lessons(id) ON DELETE CASCADE,
  teacher_id integer NOT NULL REFERENCES public.teachers(id) ON DELETE CASCADE,
  PRIMARY KEY (lesson_id, teacher_id)
);

CREATE TABLE IF NOT EXISTS public.union_groups (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  lesson_id uuid REFERENCES public.lessons(id) ON DELETE CASCADE,
  group_id integer REFERENCES public.groups(id) ON DELETE CASCADE,
  subgroup integer,
  division_id integer,
  count_divisions integer
);

CREATE INDEX IF NOT EXISTS idx_lessons_weekday
  ON public.lessons (weekday);

CREATE INDEX IF NOT EXISTS idx_lesson_teachers_teacher
  ON public.lesson_teachers (teacher_id);

CREATE INDEX IF NOT EXISTS idx_union_groups_lesson
  ON public.union_groups (lesson_id);

CREATE INDEX IF NOT EXISTS idx_union_groups_group
  ON public.union_groups (group_id);
